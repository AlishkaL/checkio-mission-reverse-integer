在不使用字符串转换的情况下，连续将给定的数字除以 10，并建立一个新的相反数字。 

def reverse_digits(num: int) -> int:

    reversed_num, absolute_num = 0, abs(num)
    while absolute_num:
        absolute_num, remainder = divmod(absolute_num, 10)
        reversed_num = reversed_num * 10 + remainder
        
    return reversed_num if num>= 0 else -reversed_num

使用两个额外函数参数的递归解决方案。 

def reverse_digits(num: int, reversed_num=0, sign=1) -> int:

    if num < 0: sign = -1
    if num == 0: return reversed_num if sign> 0 else -reversed_num
    a, b = divmod(abs(num), 10)
    
    return reverse_digits(a, reversed_num * 10 + b, sign)
